📚 自定义三维坐标系与Helmert七参数转换工具类 ​

用处说明 ​

本代码实现了一个自定义三维坐标系注册与转换工具类，支持基于Helmert七参数的三维坐标转换。它适用于地理信息系统（GIS）、三维建模、测量工程等场景，尤其是在不同坐标系之间进行精确转换、批量点转换、参数拟合和残差分析时非常有用。

主要功能 ​

  • 注册自定义坐标系：支持通过Helmert七参数注册任意三维坐标系，并自动集成到proj4投影库。
  • 坐标转换：支持WGS84经纬度与自定义坐标系之间的双向转换，支持自定义坐标系间的直接转换。
  • 批量转换：高效批量处理大量点的坐标转换。
  • 七参数拟合：可通过控制点自动拟合Helmert七参数（最小二乘法），并支持鲁棒迭代剔除粗差。
  • 残差分析：可输出每个控制点的转换残差，辅助精度评估。
  • 导入导出：支持坐标系定义的导入导出，便于项目间迁移和复用。

怎么用（使用方法示例） ​

  1. 注册自定义坐标系

    js
    import coordTool from './CustomCoordinateSystem'
coordTool.registerCoordinateSystem('MY_CRS', {
  dx: 1.2, dy: 2.3, dz: -0.5,
  rx: 0.01, ry: -0.02, rz: 0.03,
  scale: 0.000001
})

  2. WGS84经纬度转自定义坐标系

    js
    // lng, lat, height, 目标坐标系名
const xyz = coordTool.wgs84ToCustom(114.3, 30.6, 50, 'MY_CRS')

  3. 自定义坐标系转WGS84经纬度

    js
    // x, y, z, 源坐标系名
const lnglat = coordTool.customToWgs84(12345, 67890, 100, 'MY_CRS')

  4. 坐标系间直接转换

    js
    // [x, y, z], 源坐标系, 目标坐标系
const result = coordTool.transformBetweenCRS([123, 456, 789], 'MY_CRS', 'EPSG:4326')

  5. 七参数拟合与残差分析

    js
    // 控制点格式：{source: [x, y, z], target: [x, y, z]}
const params = coordTool.calculateHelmertParams(controlPoints)
const residuals = coordTool.calculateResiduals(controlPoints, params)

  6. 鲁棒拟合（自动剔除粗差）

    js
    const robustResult = coordTool.robustHelmertCalculation(controlPoints)

  7. 导入/导出坐标系定义

    js
    const def = coordTool.exportCoordinateSystem('MY_CRS')
coordTool.importCoordinateSystem(def)

适用场景 ​

  • 三维测量、工程测量、地理信息系统（GIS）数据融合
  • 多源三维数据的空间统一
  • 需要高精度坐标转换和参数拟合的工程项目

####代码

js
import * as math from 'mathjs'
import proj4 from 'proj4'

// 坐标系注册表
const coordinateSystems = {}

// 初始化基础坐标系
proj4.defs([
  ['EPSG:4326', '+title=WGS 84 (long/lat) +proj=longlat +ellps=WGS84 +datum=WGS84 +units=degrees'],
  ['EPSG:4978', '+proj=geocent +datum=WGS84 +units=m +no_defs']
])

class CustomCoordinateSystem {
  /**
   * 检查坐标点格式是否有效
   * @param {Array} point - 坐标点
   * @param {number} expectedLength - 期望的坐标维度
   * @throws {Error} 如果坐标点无效
   */
  validatePoint (point, expectedLength = 3) {
    if (!Array.isArray(point)) {
      throw new Error('坐标点必须是数组')
    }
    if (point.length < 2 || point.length > 3) {
      throw new Error('坐标点必须是2或3维数组')
    }
    if (point.some(coord => typeof coord !== 'number' || isNaN(coord))) {
      throw new Error('坐标值必须是有效数字')
    }
    if (point.length < expectedLength) {
      return point.concat(new Array(expectedLength - point.length).fill(0))
    }
    return point.slice(0, expectedLength)
  }

  /**
   * 注册自定义坐标系
   * @param {string} crsName - 坐标系名称
   * @param {Object} helmertParams - Helmert七参数
   * @param {string} [baseCRS='EPSG:4978'] - 基准坐标系
   * @throws {Error} 如果参数无效或坐标系已存在
   */
  registerCoordinateSystem (crsName, helmertParams, baseCRS = 'EPSG:4978') {
    if (!crsName || typeof crsName !== 'string') {
      throw new Error('坐标系名称必须是字符串')
    }

    if (coordinateSystems[crsName]) {
      throw new Error(`坐标系 ${crsName} 已存在`)
    }

    // 验证七参数
    const requiredParams = ['dx', 'dy', 'dz', 'rx', 'ry', 'rz', 'scale']
    for (const param of requiredParams) {
      if (typeof helmertParams[param] !== 'number' || isNaN(helmertParams[param])) {
        throw new Error(`无效的七参数: ${param}`)
      }
    }

    coordinateSystems[crsName] = {
      params: { ...helmertParams },
      baseCRS: baseCRS
    }

    // 自动定义对应的投影坐标系
    if (baseCRS === 'EPSG:4326' || baseCRS === 'EPSG:4978') {
      proj4.defs(crsName, `
              +proj=helmert
              +dx=${helmertParams.dx}
              +dy=${helmertParams.dy}
              +dz=${helmertParams.dz}
              +rx=${helmertParams.rx}
              +ry=${helmertParams.ry}
              +rz=${helmertParams.rz}
              +s=${helmertParams.scale}
              +convention=position_vector
              +units=m
              +no_defs
          `)
    }
  }

  /**
   * Helmert七参数坐标转换
   * @param {Array} xyz - 原始坐标 [x, y, z]
   * @param {Object} params - 七参数对象
   * @returns {Array} 转换后的坐标 [x, y, z]
   * @throws {Error} 如果输入无效
   */
  helmertTransform (xyz, params) {
    const [x1, y1, z1] = this.validatePoint(xyz, 3)

    const { dx, dy, dz, rx, ry, rz, scale } = params
    const rxRad = rx / 206264.80624709636 // 弧秒转弧度
    const ryRad = ry / 206264.80624709636
    const rzRad = rz / 206264.80624709636

    const x2 = dx + (1 + scale) * (x1 + rzRad * y1 - ryRad * z1)
    const y2 = dy + (1 + scale) * (-rzRad * x1 + y1 + rxRad * z1)
    const z2 = dz + (1 + scale) * (ryRad * x1 - rxRad * y1 + z1)

    return [x2, y2, z2]
  }

  /**
   * 批量坐标转换（优化性能版本）
   * @param {Array} points - 坐标数组
   * @param {Object} params - 七参数
   * @returns {Array} 转换后的坐标数组
   */
  batchTransform (points, params) {
    if (!Array.isArray(points)) {
      throw new Error('输入必须是坐标数组')
    }

    const result = new Array(points.length)
    const rxRad = params.rx / 206264.80624709636
    const ryRad = params.ry / 206264.80624709636
    const rzRad = params.rz / 206264.80624709636
    const scaleFactor = 1 + params.scale

    for (let i = 0; i < points.length; i++) {
      const [x1, y1, z1] = this.validatePoint(points[i], 3)

      result[i] = [
        params.dx + scaleFactor * (x1 + rzRad * y1 - ryRad * z1),
        params.dy + scaleFactor * (-rzRad * x1 + y1 + rxRad * z1),
        params.dz + scaleFactor * (ryRad * x1 - rxRad * y1 + z1)
      ]
    }

    return result
  }

  /**
   * 从WGS84经纬度到自定义坐标系的转换
   * @param {number} lng - 经度
   * @param {number} lat - 纬度
   * @param {number} [height=0] - 高程(米)
   * @param {string} targetCRS - 目标坐标系名称
   * @returns {Array} 目标坐标系下的[x, y, z]
   * @throws {Error} 如果坐标系未注册或输入无效
   */
  wgs84ToCustom (lng, lat, height = 0, targetCRS) {
    if (
      typeof lng !== 'number' ||
      typeof lat !== 'number' ||
      isNaN(lng) ||
      isNaN(lat)
    ) {
      throw new Error('经度和纬度必须是有效数字')
    }

    const crs = coordinateSystems[targetCRS]
    if (!crs) {
      throw new Error(`未注册的坐标系: ${targetCRS}`)
    }

    // 将WGS84经纬度转换为地心坐标
    const wgs84XYZ = proj4('EPSG:4326', 'EPSG:4978', [lng, lat, height])

    // 应用Helmert转换
    return this.helmertTransform(wgs84XYZ, crs.params)
  }

  /**
   * 从自定义坐标系到WGS84经纬度的转换
   * @param {number} x - X坐标
   * @param {number} y - Y坐标
   * @param {number} [z=0] - Z坐标
   * @param {string} sourceCRS - 源坐标系名称
   * @returns {Array} [经度, 纬度, 高程]
   * @throws {Error} 如果坐标系未注册或输入无效
   */
  customToWgs84 (x, y, z = 0, sourceCRS) {
    if (
      typeof x !== 'number' ||
      typeof y !== 'number' ||
      isNaN(x) ||
      isNaN(y)
    ) {
      throw new Error('坐标必须是有效数字')
    }

    const crs = coordinateSystems[sourceCRS]
    if (!crs) {
      throw new Error(`未注册的坐标系: ${sourceCRS}`)
    }

    // 计算反向参数
    const invParams = {
      dx: -crs.params.dx,
      dy: -crs.params.dy,
      dz: -crs.params.dz,
      rx: -crs.params.rx,
      ry: -crs.params.ry,
      rz: -crs.params.rz,
      scale: -crs.params.scale
    }

    // 反向转换
    const wgs84XYZ = this.helmertTransform([x, y, z], invParams)

    // 将地心坐标转回经纬度
    return proj4('EPSG:4978', 'EPSG:4326', wgs84XYZ)
  }

  /**
   * 坐标系间的直接转换
   * @param {Array} xyz - 原始坐标 [x, y, z]
   * @param {string} fromCRS - 源坐标系
   * @param {string} toCRS - 目标坐标系
   * @returns {Array} 转换后的坐标
   * @throws {Error} 如果坐标系未注册或输入无效
   */
  transformBetweenCRS (xyz, fromCRS, toCRS) {
    const [x, y, z] = this.validatePoint(xyz, 3)

    if (fromCRS === 'EPSG:4326' || fromCRS === 'EPSG:4978') {
      return this.wgs84ToCustom(x, y, z, toCRS)
    } else if (toCRS === 'EPSG:4326' || toCRS === 'EPSG:4978') {
      return this.customToWgs84(x, y, z, fromCRS)
    } else {
      if (!coordinateSystems[fromCRS] || !coordinateSystems[toCRS]) {
        throw new Error('源坐标系或目标坐标系未注册')
      }

      // 自定义坐标系间的转换：A→WGS84→B
      const wgs84 = this.customToWgs84(x, y, z, fromCRS)
      return this.wgs84ToCustom(...wgs84, toCRS)
    }
  }

  /**
   * 根据控制点计算Helmert七参数（最小二乘法）
   * @param {Array} controlPoints - 控制点数组
   * @returns {Object} 七参数对象
   * @throws {Error} 如果控制点不足或无效
   */
  calculateHelmertParams (controlPoints) {
    if (!Array.isArray(controlPoints)) {
      throw new Error('控制点必须是数组')
    }
    if (controlPoints.length < 3) {
      throw new Error('至少需要3个控制点')
    }

    // 构建设计矩阵A和观测向量L
    const A = []
    const L = []

    controlPoints.forEach((pt, index) => {
      try {
        const [X1, Y1, Z1] = this.validatePoint(pt.source, 3)
        const [X2, Y2, Z2] = this.validatePoint(pt.target, 3)

        A.push([1, 0, 0, 0, Z1, -Y1, X1])
        A.push([0, 1, 0, -Z1, 0, X1, Y1])
        A.push([0, 0, 1, Y1, -X1, 0, Z1])

        L.push(X2 - X1)
        L.push(Y2 - Y1)
        L.push(Z2 - Z1)
      } catch (e) {
        throw new Error(`控制点 #${index + 1} 无效: ${e.message}`)
      }
    })

    try {
      // 最小二乘解算：X = (AᵀA)⁻¹AᵀL
      const At = math.transpose(A)
      const AtA = math.multiply(At, A)
      const AtL = math.multiply(At, L)
      const X = math.lusolve(AtA, AtL)

      return {
        dx: X[0][0],
        dy: X[1][0],
        dz: X[2][0],
        rx: X[3][0] * 206264.80624709636, // 弧度转秒
        ry: X[4][0] * 206264.80624709636,
        rz: X[5][0] * 206264.80624709636,
        scale: X[6][0],
        controlPointCount: controlPoints.length
      }
    } catch (e) {
      throw new Error('参数计算失败: ' + e.message)
    }
  }

  /**
   * 计算转换残差
   * @param {Array} controlPoints - 控制点数组
   * @param {Object} params - 七参数
   * @returns {Array} 残差结果数组
   */
  calculateResiduals (controlPoints, params) {
    return controlPoints.map(pt => {
      const transformed = this.helmertTransform(pt.source, params)
      const [X2, Y2, Z2] = this.validatePoint(pt.target, 3)

      const dx = transformed[0] - X2
      const dy = transformed[1] - Y2
      const dz = transformed[2] - Z2
      const residual = Math.sqrt(dx * dx + dy * dy + dz * dz)

      return {
        point: pt,
        residual: residual,
        dx: dx,
        dy: dy,
        dz: dz
      }
    })
  }

  /**
   * 鲁棒的七参数计算（自动剔除粗差）
   * @param {Array} controlPoints - 控制点数组
   * @param {number} [threshold=0.1] - 残差阈值(米)
   * @param {number} [maxIterations=10] - 最大迭代次数
   * @returns {Object} 计算结果
   */
  robustHelmertCalculation (controlPoints, threshold = 0.1, maxIterations = 10) {
    let currentPoints = [...controlPoints]
    let bestParams = null
    let bestResiduals = []
    let iteration = 0
    let improved = true

    while (improved && iteration < maxIterations) {
      iteration++

      const params = this.calculateHelmertParams(currentPoints)
      const residuals = this.calculateResiduals(currentPoints, params)
      const rmse = Math.sqrt(
        residuals.reduce((sum, r) => sum + r.residual * r.residual, 0) /
          residuals.length
      )

      if (!bestParams || rmse < bestResiduals.rmse) {
        bestParams = params
        bestResiduals = {
          residuals: residuals,
          rmse: rmse,
          usedPoints: currentPoints.length
        }
      }

      const newPoints = currentPoints.filter((_, i) =>
        residuals[i].residual <= threshold
      )

      improved =
        newPoints.length !== currentPoints.length &&
        newPoints.length >= 3
      currentPoints = newPoints
    }

    return {
      params: bestParams,
      residuals: bestResiduals.residuals,
      rmse: bestResiduals.rmse,
      usedPointCount: bestResiduals.usedPoints,
      iterationCount: iteration
    }
  }

  /**
   * 导出坐标系定义
   * @param {string} crsName - 坐标系名称
   * @returns {Object} 坐标系定义
   */
  exportCoordinateSystem (crsName) {
    const crs = coordinateSystems[crsName]
    if (!crs) {
      throw new Error(`坐标系 ${crsName} 未注册`)
    }
    return {
      name: crsName,
      ...crs,
      proj4Definition: proj4.defs(crsName)
    }
  }

  /**
   * 导入坐标系定义
   * @param {Object} crsDef - 坐标系定义
   */
  importCoordinateSystem (crsDef) {
    if (!crsDef || !crsDef.name || !crsDef.params) {
      throw new Error('无效的坐标系定义')
    }
    this.registerCoordinateSystem(
      crsDef.name,
      crsDef.params,
      crsDef.baseCRS || 'EPSG:4978'
    )
  }
}

// 创建并导出单例
const instance = new CustomCoordinateSystem()
export default instance

📚 proj4.defs自定义坐标系参数说明 ​

1. 常用参数说明 ​

  • +proj：投影类型（如longlat、utm、geocent、helmert等）
  • +datum：大地基准（如WGS84）
  • +ellps：椭球体（如WGS84、GRS80）
  • +units：单位（如m、degrees）
  • +no_defs：不使用默认参数

2. Helmert七参数含义 ​

  • +dx：X轴平移（米）
  • +dy：Y轴平移（米）
  • +dz：Z轴平移（米）
  • +rx：X轴旋转（单位：秒，需转换为弧度，1秒=1/206264.80624709636弧度）
  • +ry：Y轴旋转（单位：秒）
  • +rz：Z轴旋转（单位：秒）
  • +s ：尺度因子（无量纲，通常为百万分之一）
  • +convention：旋转约定（如position_vector）

3. 用法示例 ​

js
proj4.defs('MY_CRS', `
  +proj=helmert
  +dx=1.2
  +dy=2.3
  +dz=-0.5
  +rx=0.01
  +ry=-0.02
  +rz=0.03
  +s=0.000001
  +convention=position_vector
  +units=m
  +no_defs
`)

4. 注意事项 ​

  • +rx、+ry、+rz为秒，内部会自动转为弧度。
  • +s为比例因子，通常为百万分之一（如0.000001表示1ppm）。
  • +convention推荐使用position_vector。
  • 注册自定义坐标系时，建议基于EPSG:4978（地心直角坐标）或EPSG:4326（经纬度）。
  • 参数顺序和拼写需严格遵循proj4规范。
