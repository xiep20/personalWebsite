  robustHelmertCalculation (controlPoints, threshold = 0.1, maxIterations = 10) {

      const residuals = this.calculateResiduals(currentPoints, params)

          residuals.length

      if (!bestParams || rmse < bestResiduals.rmse) {

        bestResiduals = {

          rmse: rmse,
          usedPoints: currentPoints.length
        }
      }

      const newPoints = currentPoints.filter((_, i) =>
        residuals[i].residual <= threshold
      )

      improved =
        newPoints.length !== currentPoints.length &&
        newPoints.length >= 3
      currentPoints = newPoints
    }

    return {
      params: bestParams,
      residuals: bestResiduals.residuals,
      rmse: bestResiduals.rmse,
      usedPointCount: bestResiduals.usedPoints,
      iterationCount: iteration
    }
  }

  /**

}
