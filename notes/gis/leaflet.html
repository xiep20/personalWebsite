调试测试自定义瓦片图层(显示行列号) ​

这个文件 TIleLayer2.js 是一个基于 Leaflet 的自定义瓦片图层（TileLayer）扩展。它的主要作用是扩展和定制 Leaflet 地图的瓦片加载和渲染方式。下面是它的核心功能说明： 自定义瓦片加载 通过重写 createTile 方法，支持自定义如何生成和加载地图瓦片（通常是图片）。可以选择返回图片元素（img），也可以返回 canvas 元素（如 getXYZ 方法所示，方便调试或自定义渲染）。 自定义投影与坐标转换 提供了 unproject2、positionToTileXY、project 等方法，支持自定义坐标投影、经纬度与像素/瓦片坐标的相互转换，适配不同的地图投影和分辨率。

js

// TileLayer2 扩展自 Leaflet 的 L.TileLayer，用于自定义瓦片图层的加载和渲染
export var TileLayer2 = L.TileLayer.extend({

    /**
     * 构造函数，初始化 TileLayer2 实例
     * @param {string} url - 瓦片服务的 URL 模板
     * @param {object} options - 配置项
     */
    initialize: function (url, options) {
        // 合并默认和传入的 options
        options = L.setOptions(this, options);
        this.url = url;
        // 调用父类的初始化方法
        L.TileLayer.prototype.initialize.call(this, this.url, options);
    },

    /**
     * 创建瓦片 DOM 元素（img 标签）
     * @param {object} coords - 瓦片坐标 {x, y, z}
     * @param {function} done - 加载完成回调
     * @returns {HTMLElement} - 瓦片图片元素
     */
    createTile: function (coords, done) {
        var tile = document.createElement('img');
        // 绑定加载和错误事件
        L.DomEvent.on(tile, 'load', L.Util.bind(this._tileOnLoad, this, done, tile));
        L.DomEvent.on(tile, 'error', L.Util.bind(this._tileOnError, this, done, tile));
        // 设置跨域属性
        if (this.options.crossOrigin || this.options.crossOrigin === '') {
            tile.crossOrigin = this.options.crossOrigin === true ? '' : this.options.crossOrigin;
        }
        // 设置 referrerPolicy
        if (typeof this.options.referrerPolicy === 'string') {
            tile.referrerPolicy = this.options.referrerPolicy;
        }
        tile.alt = '';
        // 设置图片地址
        tile.src = this.getTileUrl(coords);
        return tile;
        // 也可以返回 canvas 瓦片（调试用）
        // let tile2 = this.getXYZ(coords)
        // return tile2
    },

    /**
     * 生成 canvas 瓦片（用于调试或自定义渲染）
     * @param {object} coords - 瓦片坐标
     * @returns {HTMLCanvasElement}
     */
    getXYZ: function (coords) {
        var tile = L.DomUtil.create('canvas', 'leaflet-tile');
        var ctx = tile.getContext('2d');
        var size = this.getTileSize();
        tile.width = size.x;
        tile.height = size.y;
        // 在瓦片上绘制坐标和边框
        ctx.fillStyle = '#FFFF00';
        ctx.shadowColor = '#ffffff';
        ctx.shadowBlur = 10;
        ctx.font = "28px Arial";
        ctx.fillText('x: ' + coords.x + ', y: ' + coords.y, 30, 250);
        ctx.fillText(' zoom: ' + coords.z, 30, 290);
        ctx.fillStyle = '#67c23a';
        // ctx.fillText(nw.lng + ', ' + nw.lat, 120, 390);
        ctx.strokeStyle = 'red';
        ctx.beginPath();
        ctx.moveTo(0, 0);
        ctx.lineTo(size.x - 1, 0);
        ctx.lineTo(size.x - 1, size.y - 1);
        ctx.lineTo(0, size.y - 1);
        ctx.closePath();
        ctx.stroke();
        return tile;
    },

    /**
     * 将像素点反投影为经纬度
     * @param {L.Point} point - 像素点
     * @param {number} zoom - 缩放级别
     * @returns {L.LatLng}
     */
    unproject2: function (point, zoom) {
        return window.map.options.crs.pointToLatLng(point, zoom);
    },

    /**
     * 经纬度转瓦片坐标（支持自定义分辨率、原点等）
     * @param {object} position - 经纬度 {lng, lat}
     * @param {number} level - 缩放级别
     * @param {object} [result] - 可选，结果对象
     * @returns {L.Point}
     */
    positionToTileXY(position, level, result) {
        let _resolutions = this.options.resolutions || [];
        let _zoomOffset = this.options.zoomOffset || 0;
        let _tileSize = this.options.tileSize || 256;
        let _origin = this.options.origin || [-180, 90];
        // 检查分辨率设置
        if (!_resolutions || !_resolutions[level + _zoomOffset]) {
            return new L.Point();
        }
        // 计算当前级别的瓦片分辨率
        const tileRes = _resolutions[level + _zoomOffset] * _tileSize;
        const longitude = position.lng;
        const latitude = position.lat;
        // 计算瓦片的 X/Y 坐标
        const xTileCoordinate = Math.floor((longitude - _origin[0]) / tileRes);
        const yTileCoordinate = Math.floor((_origin[1] - latitude) / tileRes);
        if (!result) {
            return new L.Point(
                Math.max(0, xTileCoordinate),
                Math.max(0, yTileCoordinate)
            );
        }
        result.x = xTileCoordinate;
        result.y = yTileCoordinate;
        return result;
    },

    /**
     * 计算当前视图的像素范围（用于瓦片加载）
     * @param {L.LatLng} center - 地图中心点
     * @returns {L.Bounds}
     */
    _getTiledPixelBounds: function (center) {
        var map = this._map,
            mapZoom = map._animatingZoom ? Math.max(map._animateToZoom, map.getZoom()) : map.getZoom(),
            scale = map.getZoomScale(mapZoom, this._tileZoom),
            pixelCenter = this.project(center, this._tileZoom).floor(),
            // pixelCenter = map.project(center, this._tileZoom).floor(),
            halfSize = map.getSize().divideBy(scale * 2);
        // 返回像素范围
        return new L.Bounds(pixelCenter.subtract(halfSize), pixelCenter.add(halfSize));
    },

    /**
     * 坐标环绕（可自定义实现）
     * @param {object} coords - 瓦片坐标
     * @returns {object}
     */
    _wrapCoords: function (coords) {
        return coords;
    },

    /**
     * 经纬度转像素点
     * @param {L.LatLng} latlng - 经纬度
     * @param {number} zoom - 缩放级别
     * @returns {L.Point}
     */
    project: function (latlng, zoom) {
        zoom = zoom === undefined ? this._zoom : zoom;
        let crs = this.options.crs ? this.options.crs : this._map.options.crs;
        return crs.latLngToPoint(L.latLng(latlng), zoom);
    },

    /**
     * 判断瓦片坐标是否合法（是否在地图范围内）
     * @param {object} coords - 瓦片坐标
     * @returns {boolean}
     */
    _isValidTile: function (coords) {
        // let crs = this._map.options.crs;
        let crs = this.options.crs ? this.options.crs : this._map.options.crs;
        if (!crs.infinite) {
            // 如果超出范围且不支持环绕，则不加载
            var bounds = this._globalTileRange;
            if ((!crs.wrapLng && (coords.x < bounds.min.x || coords.x > bounds.max.x)) ||
                (!crs.wrapLat && (coords.y < bounds.min.y || coords.y > bounds.max.y))) { return false; }
        }
        if (!this.options.bounds) { return true; }
        // 判断瓦片是否与设定的 bounds 相交
        var tileBounds = this._tileCoordsToBounds(coords);
        return latLngBounds(this.options.bounds).overlaps(tileBounds);
    },

    /**
     * 更新各级别的瓦片容器（用于多级别切换时的 DOM 管理）
     */
    _updateLevels: function () {
        var zoom = this._tileZoom,
            maxZoom = this.options.maxZoom;
        if (zoom === undefined) { return undefined; }
        // 遍历所有级别，处理 DOM 和数据
        for (var z in this._levels) {
            z = Number(z);
            if (this._levels[z].el.children.length || z === zoom) {
                this._levels[z].el.style.zIndex = maxZoom - Math.abs(zoom - z);
                this._onUpdateLevel(z);
            } else {
                L.DomUtil.remove(this._levels[z].el);
                this._removeTilesAtZoom(z);
                this._onRemoveLevel(z);
                delete this._levels[z];
            }
        }
        var level = this._levels[zoom],
            map = this._map;
        if (!level) {
            level = this._levels[zoom] = {};
            // 创建瓦片容器
            level.el = L.DomUtil.create('div', 'leaflet-tile-container leaflet-zoom-animated', this._container);
            level.el.style.zIndex = maxZoom;
            // 记录原点
            level.origin2 = this.project(map.unproject(map.getPixelOrigin()), zoom).round();
            level.origin = map.project(map.unproject(map.getPixelOrigin()), zoom).round();
            level.zoom = zoom;
            this._setZoomTransform(level, map.getCenter(), map.getZoom());
            // 强制浏览器重新计算布局
            L.Util.falseFn(level.el.offsetWidth);
            this._onCreateLevel(level);
        }
        this._level = level;
        return level;
    },

    /**
     * 计算瓦片的像素位置（相对于当前级别的原点）
     * @param {object} coords - 瓦片坐标
     * @returns {L.Point}
     */
    _getTilePos: function (coords) {
        return coords.scaleBy(this.getTileSize()).subtract(this._level.origin2);
    },

    /**
     * 根据瓦片坐标生成瓦片图片的 URL
     * @param {object} coords - 瓦片坐标
     * @returns {string}
     */
    getTileUrl: function (coords) {
        var data = {
            r: L.Browser.retina ? '@2x' : '',
            s: this._getSubdomain(coords),
            x: coords.x,
            y: coords.y,
            z: this._getZoomForUrl()
        };
        // TMS 格式支持
        if (this._map && !this._map.options.crs.infinite) {
            var invertedY = this._globalTileRange.max.y - coords.y;
            if (this.options.tms) {
                data['y'] = invertedY;
            }
            data['-y'] = invertedY;
        }
        // 使用模板生成 URL
        return L.Util.template(this._url, L.Util.extend(data, this.options));
    },

});

        ctx.lineTo(size.x - 1, size.y - 1);
        ctx.lineTo(0, size.y - 1);

        ctx.stroke();
        return tile;
    },

    /**
     * 将像素点反投影为经纬度
     * @param {L.Point} point - 像素点
     * @param {number} zoom - 缩放级别
     * @returns {L.LatLng}

        return window.map.options.crs.pointToLatLng(point, zoom);
    },

    /**
     * 经纬度转瓦片坐标（支持自定义分辨率、原点等）
     * @param {object} position - 经纬度 {lng, lat}
     * @param {number} level - 缩放级别
     * @param {object} [result] - 可选，结果对象
     * @returns {L.Point}
     */

        let _resolutions = this.options.resolutions || [];
        let _zoomOffset = this.options.zoomOffset || 0;
        let _tileSize = this.options.tileSize || 256;
        let _origin = this.options.origin || [-180, 90];
        // 检查分辨率设置
        if (!_resolutions || !_resolutions[level + _zoomOffset]) {
            return new L.Point();
        }
        // 计算当前级别的瓦片分辨率

                zoomOffset: -6, // 瓦片的0级对应Cesium的瓦片层级，值为： 0 - Cesium层级，若瓦片的0级对应Cesium的10级，则值为 0 - 10 = -10，同时在瓦片请求时{z}的数值替换时也需加上这个层级偏移值
                tileSize: 512, // 瓦片的大小，默认为256，即一张瓦片的大小为 256 * 256
                resolutions: [0.00951784402332112, 0.00475892201166056, 0.00237946100583028, 0.00118973050291514, 0.00059486525145757, 0.000297432625728785, 0.00015228550437313792, 0.00007614275218656896, 0.00003807137609328448, 0.00001903568804664224, 0.00000951784402332112, 0.00000475892201166056], // 瓦片每一层级分辨率
                crs: L.CRS.CRS_4490
            })
            map.addLayer(layer3);
