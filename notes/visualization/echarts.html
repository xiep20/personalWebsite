饼图 Pie 系列 ​

  • pie.js
    基础饼图，适合简单占比展示。

    js
    // pie.js
/*
 * @Description: 饼图1
 */
function chartsFun (data, title, unit) {
  const dataValue = []
  let allNum = 0
  data.forEach(ele => {
    const obj = {
      name: ele.name,
      value: Number(ele.value)
    }
    dataValue.push(obj)
    allNum += obj.value
  })
  const option = {
    backgroundColor: 'transparent',
    tooltip: {
      borderWidth: 0,
      backgroundColor: 'rgba(8,36,68,.9)',
      color: '#fff',
      textStyle: {
        color: '#fff'
      },
      trigger: 'item'
    },
    legend: {
      type: 'scroll',
      orient: 'horizontal',
      top: 'bottom',
      left: '10%',
      right: '10%',
      icon: 'circle',
      textStyle: {
        color: '#fff',
        fontSize: 12
      },
      pageIconColor: '#00b4ff',
      pageTextStyle: {
        color: '#00b4ff'
      }
    },
    series: [
      {
        type: 'pie',
        radius: ['60%', '80%'],
        center: ['50%', '50%'],
        avoidLabelOverlap: false,
        itemStyle: {
          borderRadius: 10,
          borderColor: '#fff',
          borderWidth: 2,
          normal: {
            color: function (colors) {
              const colorList = [
                '#00B3FF',
                '#91DF7B',
                '#FED83A',
                '#FF9E02',
                '#49C3A4',
                '#75bedc',
                '#E690D1',
                '#FF9F7F'
              ]
              return colorList[colors.dataIndex % 8]
            }
          }
        },
        label: {
          show: true,
          fontSize: '16px',
          borderColor: '#ff0f0f',
          formatter: function (e) {
            return `${Number(allNum.toFixed(2))}${unit}\n总计`
          },
          color: '#fff',
          position: 'center'
        },
        data: dataValue
      }
    ]
  }
  return option
}

export default chartsFun

  • pie3.js
    环形饼图，适合多层级占比展示。

    js
    // pie3.js
/*
 * @Description: 环形饼图，图例右侧，支持百分比显示
 */
function chartsFun (data, title, unit) {
  const colorList = [
    '#00B3FF',
    '#91DF7B',
    '#FED83A',
    '#FF9E02',
    '#49C3A4',
    '#75bedc',
    '#E690D1',
    '#FF9F7F'
  ]
  const dataValue = data
  const option = {
    backgroundColor: 'transparent',
    tooltip: {
      trigger: 'item'
    },
    legend: {
      type: 'scroll',
      orient: 'vertical',
      right: 10,
      top: 'middle',
      left: '53%',
      icon: 'circle',
      textStyle: {
        color: '#333333',
        rich: {
          name: {
            width: 80,
            align: 'left',
            fontSize: 12,
            color: '#ffffff'
          },
          b: {
            align: 'right',
            fontSize: 12,
            color: '#ff9e02'
          }
        },
        fontWeight: 'bold',
        fontSize: 12,
        lineHeight: 26
      },
      formatter: function (name) {
        let total = 0
        let tarValue
        for (let i = 0; i < dataValue.length; i++) {
          total += dataValue[i].value
          if (dataValue[i].name === name) {
            tarValue = dataValue[i].value
          }
        }
        const p = ((tarValue / total) * 100)
        const rname = (name.length > 8 ? (name.slice(0, 8) + '...') : name)
        return '{name| ' + rname + '} ' + '{b| ' + Number(p.toFixed(2)) + '%}'
      }
    },
    series: [
      {
        name: '图斑分类',
        type: 'pie',
        radius: ['45%', '68%'],
        center: ['28%', '50%'],
        avoidLabelOverlap: false,
        itemStyle: {
          normal: {
            color: function (colors) {
              return colorList[colors.dataIndex % 8]
            }
          },
          borderRadius: 10,
          borderColor: '#fff',
          borderWidth: 2
        },
        label: {
          show: false,
          position: 'center'
        },
        emphasis: {
          label: {
            show: true,
            fontSize: 18,
            fontWeight: 'bold'
          }
        },
        labelLine: {
          show: false
        },
        data: dataValue
      }
    ]
  }
  return option
}

import * as _ from "lodash"
function chartsFun (data, title, unit) {
  const color=['#17BD62','#FFBC13','#1387FF'];
  let total = _.sumBy(data, "value");
  let data2 = data.map(i => (i.value/total*100).toFixed(2));
  let indicators = data.map(i => ({ name: i.name, max: 100 }));
  const option = {
    tooltip: { show: false, trigger: "item" },
    radar: {
      name: {
        formatter: (a, b) => {
          let i = data.findIndex(item => item.name === a);

function chartsFun (data, title, unit, echarts, myChart) {

桑基图 Sankey 系列 ​

  • sankey.js
    基础桑基图，适合能量流、资金流等关系流向展示。

    js
    // sankey.js
/*
 * @Description: 基础桑基图，适合能量流、资金流等关系流向展示
 */
function chartsFun (data, title, unit) {
  const dataN = []
  let links = []
  const colors = ['#f7a365', '#44eda1', '#00ffff', '#00baff', '#f8b551', '#7ecef4', '#7ecef4', '#7ecef4']
  data.nodes.forEach((element, index) => {
    const enode = {
      name: element.name,
      itemStyle: { normal: { color: element.color || colors[index % colors.length] } }
    }
    if (element.value) { enode.value = element.value }
    if (element.depth) { enode.depth = element.depth }
    dataN.push(enode)
  })
  links = data.links
  const option = {
    tooltip: { borderWidth: 0, backgroundColor: 'rgba(8,36,68,.9)', color: '#fff', textStyle: { color: '#fff' }, trigger: 'item', formatter: '{b}: {c}' + unit },
    series: {
      type: 'sankey', top: '10%', height: '80%', layout: 'none', emphasis: { focus: 'adjacency' },
      data: dataN, links: links,
      label: { normal: { color: '#fff', fontSize: 14 } },
      lineStyle: { normal: { color: 'source', curveness: 0.5 } },
      itemStyle: { normal: { borderWidth: 1, borderColor: 'transparent' } }
    }
  }
  return option
}
export default chartsFun
